package io.maze.objects;

import java.util.Random;

import io.maze.core.InventoryItem;

public class Locker extends Object {
    private boolean opened;
    private int containedItem; //apple or cookie or rotten apple

    public Locker(String name) {
        super(name, "locker.png");
        this.opened = false;

        //special case for locker8 in the hidden room - always contains the student card
        if (name.equals("locker8")) {
            containedItem = InventoryItem.STUDENT_CARD;
        } else {
            //normal locker contents logic
            Random random = new Random();
            float chance = random.nextFloat(); //0.0 to 1.0
            if (chance < 0.5f) { //50% chance for the apple
                containedItem = InventoryItem.APPLE;
            } else if (chance < 0.8f) { //30% chance for cookie (0.5 to 0.8)
                containedItem = InventoryItem.COOKIE;
            }else {
                containedItem = InventoryItem.BAD_APPLE;
            }
        }

    }

    public int open() {
        if (!opened) {
            opened = true;
            return containedItem;
        }
        return InventoryItem.EMPTY;
    }

    public boolean isOpened() {
        return opened;
    }

    public void dispose() {
        super.dispose();
    }
}
