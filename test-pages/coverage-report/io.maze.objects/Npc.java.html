package io.maze.objects;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector2;
import io.maze.entities.Player;

/**
 * Represents the stationary NPCs which each have a single texture
 * The NPC does not move and can be interacted with by the player
 */
public class Npc extends Object {
    private String dialogue;
    private boolean showMessage = false;
    private BitmapFont font;
    private Vector2 position;
    private Texture tecture;

    /**
     * Creates a new stationary NPC with its corresponding single texture
     * @param texturePath the path to the NPC's texture
     * @param name the name of the NPC object in the Tiled map
     * @param dialogue the dialogue text the NPC will say 
     */
    
    public Npc(String name, String texturePath, String dialogue) {
        //pass the same texture path for all directions as the NPC is stationary and doesn't move
        super(name, texturePath);
        this.dialogue = dialogue;
        this.font = new BitmapFont();
        this.font.getData().setScale(0.025f); 
        this.font.setUseIntegerPositions(false);
    }

    /**
     * Update if the dialogue is shown to the player depedning on the player's position anf if E has just been pressed
     * @param player Player object
     */

    public void update(Player player) {
        // Calculate distance between player and NPC
        float playerX = player.getX();
        float playerY = player.getY();
        float npcX = getSprite().getX();
        float npcY = getSprite().getY();
        
        float distanceX = Math.abs(playerX - npcX);
        float distanceY = Math.abs(playerY - npcY);
        float distance = (float) Math.sqrt(distanceX * distanceX + distanceY * distanceY);

        if (distance < 2f && Gdx.input.isKeyJustPressed(Input.Keys.E)) {
            showMessage = true;
        }

        if(showMessage && distance > 3f) {
            showMessage = false;
        }
    }

    /**
     * Convenience method to be called by the game screen's renderer() method to draw the NPC's dialogue above ir
     * @param batch SpriteBatch used by application to render all sprites
     *
     */
    public void renderDialogue(SpriteBatch batch) {
        if (showMessage) {
            font.draw(
                batch,
                dialogue,
                getSprite().getX() -2f,
                getSprite().getY() + getSprite().getHeight() + 1f);
        }
    }

    public void dispose() {
        super.dispose();
        if (font != null) {
            font.dispose();
        }
    }
    
    public String getDialogue() {
        return dialogue;
    }


    public void setDialogue(String dialogue) {
        this.dialogue = dialogue;
    }

    /**
     * Get the NPC's position in the world
     * @return the NPC's x-by-y coordinates as a 2D vector
     */
    public Vector2 getPosition() {
        return new Vector2(getSprite().getX(),getSprite().getY());
    }

}
