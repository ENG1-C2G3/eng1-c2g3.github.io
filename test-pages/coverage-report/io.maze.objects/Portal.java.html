package io.maze.objects;

/**
 * Represents a portal that transports the player to a different map
 * The player is taken to a new map when they press E over the portal 
 */
public class Portal extends Object{
    private final String targetMap;
    private boolean activated;

    public Portal(String name, String targetMap) {
        super(name, "objects/transparent.png");
        this.targetMap = targetMap;
        this.activated = false;
    }

    public String getTargetMap() {
        return targetMap;
    }

    public boolean isActivated() {
        return activated;
    }

    public void setActivated(boolean activated) {
        this.activated = activated;
    }
    
    public void dispose() {
        //nothing to dispose atm
    }
}
