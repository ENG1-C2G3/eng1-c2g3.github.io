package io.maze.screens;

//when player clicks e, open a screen where the player can click buttons (numbers 0-9) and input three 3 numbers then click enter, if the correct code 793 is enterred the user can walk through lockedDoor collision layer 
import java.util.ArrayList;
import java.util.List;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.InputProcessor;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector3;
import com.badlogic.gdx.utils.ScreenUtils;
import com.badlogic.gdx.utils.viewport.ExtendViewport;

import io.maze.core.Main;

/**
 * Keypad screen that allows the player to input a 3-digit code
 * If correct code (793) is entered, the locked door collision is removed
 */

public class KeypadScreen implements Screen, InputProcessor {
    private final Main game;
    private final GameScreen gameScreen;
    private final ExtendViewport viewport;
    private BitmapFont font;
    private final SpriteBatch batch;

    private List<Integer> enteredCode;
    private boolean codeCorrect;
    private String statusMessage;

    //textures for the buttons
    private Texture buttonTexture;
    private Texture buttonPressedTexture;

    public KeypadScreen(final Main game, final GameScreen gameScreen) {
        this.game = game;
        this.gameScreen = gameScreen;
        this.batch =game.getBatch();
        this.viewport = new ExtendViewport(16, 9);
        this.font = new BitmapFont();
        font.setUseIntegerPositions(false);
        font.getData().setScale(0.05f);
        font.setColor(Color.WHITE);

        this.enteredCode = new ArrayList<>();
        this.codeCorrect = false;
        this.statusMessage = "Enter a 3-digit code";

        //load button textures -> need to create them
        this.buttonTexture = new Texture("objects/buttonInactive.png"); //default greybutton atm
        this.buttonPressedTexture = new Texture("objects/buttonActive.png"); //pressed state

        //set this screen as the input processor
        Gdx.input.setInputProcessor(this);

    }

    @Override
    public void render(float delta) {
        ScreenUtils.clear(Color.DARK_GRAY);
        viewport.apply();
        batch.setProjectionMatrix(viewport.getCamera().combined);
        batch.begin();

        //draw the title
        font.draw(batch, "SECURITY KEYPAD",4.6f, 10.8f);
        font.draw(batch, statusMessage, 5.4f, 8.5f);

        //draw the entered code
        StringBuilder codeDisplay = new StringBuilder();
        for (int i = 0; i < 3; i++) {
            if (i < enteredCode.size()) {
                codeDisplay.append(enteredCode.get(i));
            }else {
                codeDisplay.append("_");
            }
            if (i < 2) {
                codeDisplay.append(" ");
            }
        }
        font.draw(batch, "Code: " + codeDisplay.toString(), 1.5f, 6.4f);

        //draw the keypad buttons (3x4 grid 0-9)
        float buttonSize = 1.2f;
        float startX = 6f;
        float startY = 6f;
        float spacing = 0.2f;

        //number 0-9
        String[] labels = {"1", "2", "3", "4", "5", "6", "7", "8", "9", "C", "0", "E"};

        for (int row = 0; row < 4; row ++) {
            for (int col = 0; col < 3; col ++) {
                int index = row * 3 + col;
                if (index < labels.length) {
                    float x = startX + col * (buttonSize + spacing);
                    float y = startY - row * (buttonSize + spacing);

                    batch.draw(buttonTexture, x, y, buttonSize, buttonSize);
                    font.draw(batch, labels[index], x + buttonSize/3, y + buttonSize/1.5f);
    
                }
            }
        }

        //if the entered code is correc show a success message and an option to close
        if (codeCorrect) {
            font.setColor(Color.GREEN);
            font.draw(batch, "ACCESS GRANTED!", 4.4f, 1.5f);
            font.setColor(Color.WHITE);
        }

        batch.end();
    }

    private void handleButtonPress(String button) {
        if (codeCorrect) return; //don't accept input after the code has already been entered correctly
        switch(button) {
            case "C": //clear
                enteredCode.clear();
                statusMessage = "Code cleared";
                break;
            case "E": //enter code
                checkCode();
                break;
            default: //number
                if (enteredCode.size() < 3 && button.matches("\\d")) {
                    enteredCode.add(Integer.parseInt(button));
                    statusMessage = "Enter 3-digit code";

                    //auto submit when 3 digits are entered
                    if (enteredCode.size() == 3) {
                        checkCode();
                    }
                }
                break;
        }
    }

    private void checkCode() {
        if (enteredCode.size() != 3) {
            statusMessage = "Enter exactly 3 digits";
            return;
        }

        //check if the code is 793
        if (enteredCode.get(0) == 7 && enteredCode.get(1) == 9 && enteredCode.get(2) == 3) {
            codeCorrect = true;
            statusMessage = "Code correct!";

            //remove the locked door from collision -> need to implement in game screen
            if (gameScreen != null) {
                gameScreen.unlockDoor();
            }

            game.setScreen(gameScreen);
            gameScreen.unPause();
        }else {
            statusMessage = "Incorrect code!";
            enteredCode.clear();
        }
    }

    public boolean isCodeCorrect() {
        return codeCorrect;
    }

    //InputProcessor methods for mouse/touch input
    @Override
    public boolean touchDown(int screenX, int screenY, int pointer, int button) {
        Vector3 touchPos = new Vector3(screenX, screenY, 0);
        viewport.getCamera().unproject(touchPos);

        //button gid coordinates (same as in ther render method)
        float buttonSize = 1.2f;
        float startX = 6.f;
        float startY = 6f;
        float spacing = 0.2f;

        String[] labels = {"1", "2", "3", "4", "5", "6", "7", "8", "9", "C", "0", "E"};

        for (int row = 0; row < 4; row ++) {
            for (int col = 0; col < 3; col ++) {
                int index = row * 3 + col;
                if (index < labels.length) {
                    float x = startX + col * (buttonSize + spacing);
                    float y = startY - row * (buttonSize + spacing);

                    if (touchPos.x >= x && touchPos.x <= x + buttonSize && touchPos.y >= y && touchPos.y <= y + buttonSize) {
                        handleButtonPress(labels[index]);
                        return true;
                    }

                }  
            }
        }
        return false;
    }

    //required screen methods
    @Override
    public void show() {}

    @Override
    public void resize(int width, int height) {
        viewport.update(width, height);
        viewport.getCamera().position.set(viewport.getWorldWidth() / 2f, viewport.getWorldHeight() / 2f, 0);
        viewport.getCamera().update();
    }

    @Override
    public void pause() {}

    @Override
    public void resume() {}

    @Override
    public void hide() {
        Gdx.input.setInputProcessor(null); //reset the input processsor
    }

    @Override
    public void dispose() {
        font.dispose();
        buttonTexture.dispose();
        buttonPressedTexture.dispose();
    }

    //other InputProcessor methods (required but they aren't used)
    @Override public boolean keyDown(int keycode) {
        //close the screen when the player presses ESC
        if (keycode == com.badlogic.gdx.Input.Keys.ESCAPE) {
            game.setScreen(gameScreen);
            gameScreen.unPause();
            dispose();
            return true;
        }
        return false;
    }
    @Override public boolean keyUp(int keycode) {return false;}
    @Override public boolean keyTyped(char character) {return false;}
    @Override public boolean touchUp(int screenX, int screenY, int pointer, int button) {return false;}
    @Override public boolean touchDragged(int screenX, int screenY, int pointer) {return false;}
    @Override public boolean mouseMoved(int screenX, int screenY) {return false;}
    @Override public boolean scrolled(float amountX, float amountY) {return false;}
    @Override public boolean touchCancelled(int screenX, int screenY, int pointer, int button) {return false;}
}
