package io.maze.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.math.Vector3;
import com.badlogic.gdx.utils.ScreenUtils;
import com.badlogic.gdx.utils.viewport.ExtendViewport;
import io.maze.core.LeaderboardManager;
import io.maze.core.Main;
import io.maze.core.ScoreEntry;

import java.util.List;

/**
 * Displays the locally stored leaderboard.
 */
public class LeaderboardScreen implements Screen {
    private final Main game;
    private final ExtendViewport viewport;
    private final BitmapFont font;
    private final LeaderboardManager leaderboardManager;
    private final Texture backActive;
    private final Texture backInactive;

    public LeaderboardScreen(final Main game) {
        this.game = game;
        this.viewport = new ExtendViewport(16, 9);
        this.font = new BitmapFont();
        font.setUseIntegerPositions(false);
        font.getData().setScale(0.07f);
        this.leaderboardManager = game.getLeaderboardManager();
        this.backActive = new Texture("menu/backActive.png");
        this.backInactive = new Texture("menu/backInactive.png");
    }

    @Override
    public void render(float delta) {
        ScreenUtils.clear(Color.BLACK);

        viewport.apply();
        game.getBatch().setProjectionMatrix(viewport.getCamera().combined);
        game.getBatch().begin();

        font.draw(game.getBatch(), "Leaderboard", 5.4f, 11.f);

        List<ScoreEntry> scores = leaderboardManager.getScores();
        if (scores.isEmpty()) {
            font.draw(game.getBatch(), "No scores yet.", 4.8f, 7.5f);
        } else {
            for (int i = 0; i < scores.size(); i++) {
                ScoreEntry entry = scores.get(i);
                String entryName = entry.name == null ? "" : entry.name.trim();
                if (entryName.isEmpty()) {
                    entryName = "Player";
                }
                font.draw(game.getBatch(), (i + 1) + ". " + entryName + " - " + entry.score, 4.5f, 9.5f - i * 1f);
            }
        }

        float buttonWidth = 5f;
        float buttonHeight = 2f;
        float buttonX = viewport.getWorldWidth() / 2f - buttonWidth / 2f;
        float buttonY = 1f;

        Vector3 touchPos = new Vector3(Gdx.input.getX(), Gdx.input.getY(), 0f);
        viewport.getCamera().unproject(touchPos);

        boolean hoveringBack = touchPos.x > buttonX && touchPos.x < buttonX + buttonWidth
            && touchPos.y > buttonY && touchPos.y < buttonY + buttonHeight;

        game.getBatch().draw(hoveringBack ? backActive : backInactive, buttonX, buttonY, buttonWidth, buttonHeight);

        if (hoveringBack && Gdx.input.justTouched()) {
            game.setScreen(new MainMenu(game));
            dispose();
        }

        game.getBatch().end();
    }

    @Override
    public void show() {
        // no-op
    }

    @Override
    public void resize(int width, int height) {
        viewport.update(width, height, true);
    }

    @Override
    public void pause() {
        // no-op
    }

    @Override
    public void resume() {
        // no-op
    }

    @Override
    public void hide() {
        // no-op
    }

    @Override
    public void dispose() {
        font.dispose();
        backActive.dispose();
        backInactive.dispose();
    }
}
